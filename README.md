// 1) Найди нужный path.
// Часто достаточно первого:
const paths = [...document.querySelectorAll('path')];
console.log('paths:', paths.length);
const path = paths[0]; // если не тот — выбери другой индекс

// 2) Настройки семплирования
const step = 5; // пикселей вдоль линии между точками (5–10 для MVP норм)

// 3) Сэмплирование
const len = path.getTotalLength();
const pts = [];
for (let d = 0; d <= len; d += step) {
  const p = path.getPointAtLength(d);
  pts.push({ x: p.x, y: p.y });
}

// 4) Если нужно замкнуть контур:
if (pts.length > 0) pts.push({ ...pts[0] });

// 5) Вывод
console.log('len:', len, 'points:', pts.length);
copy(JSON.stringify(pts));


// Normalized
const W = 1365; // ширина PNG/Frame
const H = 1005; // высота PNG/Frame
const norm = pts.map(p => ({ x: p.x / W, y: p.y / H }));
copy(JSON.stringify(norm));
